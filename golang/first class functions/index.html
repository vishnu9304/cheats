    first class functions

    What are first class functions?

    A language which supports first class functions allows functions to be assigned to variables, passed as arguments to other functions and returned from other functions.

    Anonymous functions

    Example:

    package main

import "fmt"

func main() {
    a := func(a, b int) int {
        return a + b
    }
    x := a(10, 20)
    fmt.Println(x)

    // Calling Anonymous function without assigning it to variable.

    func(a, b int) {
        fmt.Println(a + b)
    }(10, 20)
}

    User defined function types

    Example:

    package main

import "fmt"

type add func(a, b int) int

func main() {
    var a add = func(a, b int) int {
        return a + b
    }
    a(10, 20)
    fmt.Printf("%T\n", a)
}

    Higher order functions

    It does any one of the following.

    1. Takes one or more functions as arguments.
    2. Returns a function as its result.

    Passing functions as arguments to other functions

    Example:

    package main

import "fmt"

// exec received a function of type **func(a, b int)**

func exec(f func(a, b int) int, a, b int) int {
    return f(a, b)
}

func main() {
    a := func(a, b int) int {
        return a + b
    }
    x := exec(a, 10, 20)
    fmt.Println(x)
}

    Returning functions from other functions

    Example:

    package main

import "fmt"

func returnSomeFunction() func(a, b int) int {
    f := func(a, b int) int {
        return a + b
    }
    return f
}

func main() {
    f := returnSomeFunction()
    fmt.Println(f(10, 20))
}

    Closures

    Closures are anonymous functions which access the variables defined outside the body of the function.

    Example:

    package main

import "fmt"

func main() {
    t := "Hello!"
    f := func(s string) string {
        return t + " " + s
    }
    fmt.Println(f("Vishnu"))
}

    Final Example

    package main

import "fmt"

type student struct {
    name     string
    location string
    grade    string
}

func gradeFilter(students []student, f func(student) bool) []student {
    var s []student
    for _, student := range students {
        if f(student) == true {
            s = append(s, student)
        }
    }
    return s
}

func main() {
    s1 := student{
        name:     "vishnu",
        location: "USA",
        grade:    "A+",
    }
    s2 := student{
        name:     "Pradeep",
        location: "papanaykenpatti",
        grade:    "C+",
    }
    s := []student{s1, s2}
    f := func(s student) bool {
        if s.grade == "C+" {
            return true
        }
        return false
    }
    fmt.Println(gradeFilter(s, f))
}
